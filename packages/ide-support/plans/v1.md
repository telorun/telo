# @telorun/ide-support — v1

Completions + diagnostic normalization for Telo manifests, shared by the VS Code extension and the telo-editor Monaco source tab.

### apps/telo-editor/src/components/views/source/register-completion.ts

New module-level Monaco provider registration + live registry accessor:

```ts
import type * as monacoNs from "monaco-editor";
import type { AnalysisRegistry } from "@telorun/analyzer";
import { buildCompletions } from "@telorun/ide-support";

let registered = false;
const registryRef: { current: AnalysisRegistry | undefined } = { current: undefined };

export function setActiveRegistry(r: AnalysisRegistry | undefined) {
  registryRef.current = r;
}

export function registerYamlCompletions(monaco: typeof monacoNs) {
  if (registered) return;
  registered = true;
  monaco.languages.registerCompletionItemProvider("yaml", {
    triggerCharacters: [" ", ":"],
    provideCompletionItems(model, position) {
      const word = model.getWordUntilPosition(position);
      const range = {
        startLineNumber: position.lineNumber,
        endLineNumber: position.lineNumber,
        startColumn: word.startColumn,
        endColumn: word.endColumn,
      };
      const results = buildCompletions(
        model.getValue(),
        position.lineNumber - 1,
        position.column - 1,
        registryRef.current,
      );
      const kindMap = {
        class: monaco.languages.CompletionItemKind.Class,
        enumMember: monaco.languages.CompletionItemKind.EnumMember,
        property: monaco.languages.CompletionItemKind.Property,
      };
      return {
        suggestions: results.map((r) => ({
          label: r.label,
          kind: kindMap[r.kind],
          detail: r.detail,
          documentation: r.documentation,
          insertText: r.insertText ?? r.label,
          insertTextRules: r.snippet
            ? monaco.languages.CompletionItemInsertTextRule.InsertAsSnippet
            : undefined,
          sortText: r.sortText,
          preselect: r.preselect,
          range,
        })),
      };
    },
  });
}
```

Registration is language-scoped and happens once; the ref indirection keeps the provider reading the freshest registry across every analysis pass.

## Changed files

### ide/vscode/src/completion.ts

Replace the entire file with a thin adapter:

```ts
import { buildCompletions, type CompletionResult } from "@telorun/ide-support";
import type { AnalysisRegistry } from "@telorun/analyzer";
import * as vscode from "vscode";

const KIND_MAP: Record<CompletionResult["kind"], vscode.CompletionItemKind> = {
  class: vscode.CompletionItemKind.Class,
  enumMember: vscode.CompletionItemKind.EnumMember,
  property: vscode.CompletionItemKind.Property,
};

function toItem(r: CompletionResult): vscode.CompletionItem {
  const item = new vscode.CompletionItem(r.label, KIND_MAP[r.kind]);
  if (r.detail) item.detail = r.detail;
  if (r.documentation) item.documentation = r.documentation;
  if (r.insertText !== undefined) {
    item.insertText = r.snippet ? new vscode.SnippetString(r.insertText) : r.insertText;
  }
  if (r.preselect) item.preselect = true;
  if (r.sortText) item.sortText = r.sortText;
  return item;
}

export class TeloCompletionProvider implements vscode.CompletionItemProvider {
  private readonly registries = new Map<string, AnalysisRegistry>();
  updateRegistry(filePath: string, registry: AnalysisRegistry) { this.registries.set(filePath, registry); }
  deleteRegistry(filePath: string) { this.registries.delete(filePath); }
  provideCompletionItems(document: vscode.TextDocument, position: vscode.Position) {
    if (document.languageId !== "yaml") return;
    const registry = this.registries.get(document.uri.fsPath);
    return buildCompletions(document.getText(), position.line, position.character, registry).map(toItem);
  }
}
```

### apps/telo-editor/src/components/views/source/SourceView.tsx

Three distinct additions:

1. **Register the Monaco completion provider once.** In the first `onMount` callback ([lines 306-321](apps/telo-editor/src/components/views/source/SourceView.tsx#L306-L321)), call `registerYamlCompletions(monaco)`. The function is idempotent; subsequent mounts return early.

2. **Analyzer-diagnostic marker effect.** New `useEffect` keyed on `(workspaceDiagnostics, sourceFiles)`. Iterate every file with an open model; for each diagnostic, resolve the owning manifest via `workspaceDiagnostics.manifestsByResource` to pull `positionIndex`/`sourceLine`:

   ```ts
   const diags = [
     ...(workspaceDiagnostics.byFile.get(filePath) ?? []),
     ...Array.from(workspaceDiagnostics.byResource.get(filePath)?.values() ?? []).flat(),
   ];
   const model = editorsRef.current[filePath]?.getModel();
   if (!model) continue;
   const markers = diags
     .map((d) => {
       const name = (d.data as any)?.resource?.name as string | undefined;
       const m = name
         ? workspaceDiagnostics.manifestsByResource.get(`${filePath}::${name}`)
         : undefined;
       return normalizeDiagnostic(d, {
         registry: workspaceDiagnostics.registry,
         positionIndex: (m?.metadata as any)?.positionIndex,
         sourceLine: (m?.metadata as any)?.sourceLine,
       });
     })
     .map((n) => toMonacoMarker(n, monacoRef.current!));
   monaco.editor.setModelMarkers(model, "telo", markers);
   ```

   Diagnostics routed to `byFile` with no `data.resource` fall through to the `(0,0)-(0,0)` leg of `resolveRange`. This is **separate from `setMarker`**; it does not modify the existing `commit()` path.

3. **Keep existing parse-error path untouched.** The current `setMarker` call at [lines 188-208](apps/telo-editor/src/components/views/source/SourceView.tsx#L188-L208) writes under marker owner `"yaml-parse"`. The new analyzer markers write under owner `"telo"`. Monaco composes markers from multiple owners per model, so both coexist and can be cleared independently.

### analyzer/nodejs/src/kind-suggest.ts (new)

Private algorithm module keyed on `(aliases, defs)` so it is callable from both the stateful `AnalysisRegistry` API and the stateless `analyze()` path (the CLI flow hits the latter — [analyzer.ts:288](analyzer/nodejs/src/analyzer.ts#L288) sets `ctx = registry?._context()`, so `registry` is optional).

```ts
export function computeValidUserFacingKinds(
  aliases: AliasResolver,
  defs: DefinitionRegistry,
): string[];

export function computeSuggestKind(
  badKind: string,
  aliases: AliasResolver,
  defs: DefinitionRegistry,
): string | undefined;
```

`computeValidUserFacingKinds` implements the logic currently inlined at [ide/vscode/src/completion.ts:233-255](ide/vscode/src/completion.ts#L233-L255): the hardcoded Telo root kinds plus, for every non-abstract registered definition, the alias form `${alias}.${TypeName}` for each alias pointing at its module.

`computeSuggestKind` runs Levenshtein over that list, threshold `min(3, floor(badKind.length / 3))`, case-sensitive, undefined on ties.

Not exported from the package index — consumers go through `AnalysisRegistry` (IDE hosts) or the in-`analyze` direct call (CLI).
